# DOCX

`python-docx` is the standard library for reading and writing `.docx` files.
Always import it as `from docx import Document`.

## Read an existing document

```python
from docx import Document

doc = Document("file.docx")

# All paragraphs
for para in doc.paragraphs:
    print(para.style.name, "|", para.text)

# Tables
for table in doc.tables:
    for row in table.rows:
        for cell in row.cells:
            print(cell.text, end="\t")
        print()
```

Access the full text in one shot:
```python
full_text = "\n".join(p.text for p in doc.paragraphs)
```

## Create a document from scratch

```python
from docx import Document
from docx.shared import Pt, Inches, RGBColor

doc = Document()

# Title
doc.add_heading("Report Title", level=0)

# Section heading + paragraph
doc.add_heading("Introduction", level=1)
doc.add_paragraph("This document was generated by Nexus.")

# Formatted run
para = doc.add_paragraph()
run = para.add_run("Bold and coloured text.")
run.bold = True
run.font.color.rgb = RGBColor(0x1F, 0x49, 0x7D)
run.font.size = Pt(12)

# Page break
doc.add_page_break()

doc.save("output.docx")
```

## Styles

Apply a named style to a paragraph:
```python
para = doc.add_paragraph("Intro text", style="Body Text")
doc.add_heading("Section", level=2)   # Heading 2 style
```

List all available styles:
```python
for style in doc.styles:
    if style.type.name == "PARAGRAPH":
        print(style.name)
```

## Tables

```python
from docx import Document
from docx.shared import Pt

doc = Document()
table = doc.add_table(rows=1, cols=3)
table.style = "Table Grid"

# Header row
hdr = table.rows[0].cells
hdr[0].text = "Name"
hdr[1].text = "Role"
hdr[2].text = "Score"

# Data rows
data = [("Alice", "Engineer", "95"), ("Bob", "Designer", "88")]
for name, role, score in data:
    row = table.add_row().cells
    row[0].text = name
    row[1].text = role
    row[2].text = score

doc.save("table.docx")
```

## Headers and footers

```python
from docx import Document
from docx.shared import Pt

doc = Document()
section = doc.sections[0]

# Header
header = section.header
header.paragraphs[0].text = "Confidential — Nexus Report"

# Footer with page number
from docx.oxml.ns import qn
from docx.oxml import OxmlElement

footer_para = section.footer.paragraphs[0]
footer_para.text = "Page "
run = footer_para.add_run()
fldChar = OxmlElement("w:fldChar")
fldChar.set(qn("w:fldCharType"), "begin")
run._r.append(fldChar)
instrText = OxmlElement("w:instrText")
instrText.text = "PAGE"
run._r.append(instrText)
fldChar2 = OxmlElement("w:fldChar")
fldChar2.set(qn("w:fldCharType"), "end")
run._r.append(fldChar2)

doc.save("with_header_footer.docx")
```

## Edit an existing document

Read, modify, save — never mutate in-place without a backup:

```python
from docx import Document
import shutil

shutil.copy("original.docx", "original.docx.bak")

doc = Document("original.docx")

# Find and replace text in paragraphs
for para in doc.paragraphs:
    if "OLD_TEXT" in para.text:
        for run in para.runs:
            run.text = run.text.replace("OLD_TEXT", "NEW_TEXT")

# Find and replace in tables
for table in doc.tables:
    for row in table.rows:
        for cell in row.cells:
            for para in cell.paragraphs:
                for run in para.runs:
                    run.text = run.text.replace("OLD_TEXT", "NEW_TEXT")

doc.save("original.docx")
```

## Table of contents

`python-docx` can insert a TOC field but Word must refresh it on open.
Insert the field:
```python
from docx.oxml.ns import qn
from docx.oxml import OxmlElement

doc = Document()
doc.add_heading("Contents", level=1)

para = doc.add_paragraph()
run = para.add_run()
fldChar = OxmlElement("w:fldChar")
fldChar.set(qn("w:fldCharType"), "begin")
run._r.append(fldChar)

instrText = OxmlElement("w:instrText")
instrText.set(qn("xml:space"), "preserve")
instrText.text = 'TOC \\o "1-3" \\h \\z \\u'
run._r.append(instrText)

fldChar2 = OxmlElement("w:fldChar")
fldChar2.set(qn("w:fldCharType"), "separate")
run._r.append(fldChar2)

fldChar3 = OxmlElement("w:fldChar")
fldChar3.set(qn("w:fldCharType"), "end")
run._r.append(fldChar3)
```

Tell the user: "The TOC will appear after you open the file and accept the 'Update fields' prompt in Word or LibreOffice."

## Document properties

```python
from docx import Document

doc = Document("file.docx")
props = doc.core_properties
print(props.title, props.author, props.created)

props.title = "New Title"
props.author = "Nexus"
doc.save("file.docx")
```
